# quarta_aula-curso-Guanabara-
Quarta aula do curso do Guanabara em JS &lt; JavaScript 

```

<!DOCTYPE html>
<html>

<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1">
  <title>quarta aula do curso do Guanabara de JavaScript</title>
</head>

<body>
<script>
  
  var nome = window.prompt("Digite o seu nome ")
  
  /* ######### <---o document.write ele escreve na tela --->*/
  document.write(`o seu nome tem ${nome.length} letras <br> ` ) //vendo quantas letras o meu nome tem 
  document.write(`o nome ${nome} em maiúsculo é , ${nome.toUpperCase()} `) // jogando a letra para maiúsculo
  document.write(`<br> e o seu nome em minúsculo é , ${nome.toLowerCase()}`)//jogando o nome para minúsculo 
  
  
  
</script>
</body>

</html>

```
